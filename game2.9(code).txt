function addParticles(x, y, color, count=8) {
  for (let i=0;i<count;i++) {
    const a = Math.random()*Math.PI*2, spd = 1+Math.random()*4;
    particles.push({ x, y, vx:Math.cos(a)*spd, vy:Math.sin(a)*spd, life:1, color });
  }
}

// ═══ COLLISION ═══════════════════════════════════════════════════
function dist2(a,b){ const dx=a.x-b.x,dy=a.y-b.y; return dx*dx+dy*dy; }
function distCC(a,b){ return Math.sqrt(dist2(a,b)); }
function cirHit(a,b){ return distCC(a,b) < a.size+b.size; }

const MSEP = 18;
function resolveMonsterCollisions() {
  for (let i=0;i<monsters.length;i++) for (let j=i+1;j<monsters.length;j++) {
    const m1=monsters[i],m2=monsters[j];
    const dx=m2.x-m1.x,dy=m2.y-m1.y;
    const d=Math.sqrt(dx*dx+dy*dy)||0.01;
    const minD=m1.size+m2.size+MSEP;
    if (d<minD) {
      const ov=minD-d, nx=dx/d, ny=dy/d;
      m1.x-=nx*ov*0.5; m1.y-=ny*ov*0.5;
      m2.x+=nx*ov*0.5; m2.y+=ny*ov*0.5;
      const dvx=m2.vx-m1.vx,dvy=m2.vy-m1.vy, vel=dvx*nx+dvy*ny;
      if (vel>0) continue;
      const j2=-(1+1.4)*vel/2;
      m1.vx-=j2*nx; m1.vy-=j2*ny;
      m2.vx+=j2*nx; m2.vy+=j2*ny;
      for (const m of [m1,m2]) {
        const s=Math.sqrt(m.vx**2+m.vy**2), cap=m.speed*2.5;
        if (s>cap){ m.vx=m.vx/s*cap; m.vy=m.vy/s*cap; }
      }
    }
  }
}

// ═══ PLAYER HIT ══════════════════════════════════════════════════
function playerHit(p) {
  if (shieldActive && p===player) {
    shieldActive=false; shieldCooldown=true;
    addParticles(p.x,p.y,'#00ccff',20);
    const dx=p.x-(p.hitMonster?.x||p.x), dy=p.y-(p.hitMonster?.y||p.y);
    const d=Math.sqrt(dx*dx+dy*dy)||1;
    p.x=Math.max(p.size,Math.min(W-p.size,p.x+(dx/d)*70));
    p.y=Math.max(p.size,Math.min(H-p.size,p.y+(dy/d)*70));
    return;
  }
  gameOver();
}

// ═══ UPDATE ══════════════════════════════════════════════════════
function update(ts) {
  if (!gameRunning||isPaused) return;
  const now = ts;

  // Арена: сужаем
  if (currentMode==='arena' && arenaRadius>0) {
    arenaRadius = Math.max(180, arenaRadius - 0.04);
    arenaWall.style.display='block';
    monsters.forEach(m => {
      const dx=m.x-W/2,dy=m.y-H/2;
      if (Math.sqrt(dx*dx+dy*dy)>arenaRadius+60) { const p=rndInside(100); m.x=p.x; m.y=p.y; }
    });
    const pd=Math.sqrt((player.x-W/2)**2+(player.y-H/2)**2);
    if (pd>arenaRadius-player.size) playerHit(player);
  }

  if (decoy && now>decoy.until) decoy=null;
  if (combo>0 && now-comboTimer>COMBO_TIMEOUT) { combo=0; comboEl.style.display='none'; }

  if (!blueOrb.active && now-blueOrb.lastSpawnTime>blueOrb.spawnInterval) spawnBlueOrb(now);
  if (blueOrb.active && now-blueOrb.spawnedAt>blueOrb.lifeTime) { blueOrb.active=false; blueOrb.lastSpawnTime=now; }

  const specGems = [goldGem,poisGem,iceGem,magnetGemObj];
  specGems.forEach(g => {
    if (!g.active && now-g.lastSpawn>g.spawnInterval) { spawnSpecialGem(g); g.lastSpawn=now; g.spawnedAt=now; }
    if (g.active && now-g.spawnedAt>10000) { g.active=false; g.lastSpawn=now; }
  });

  // Dash
  if (keys.Space && !dashActive && now-dashCooldown>DASH_CD) {
    dashActive=true; dashTimer=now; dashCooldown=now;
    let mx=0,my=0;
    if (keys.KeyW||keys.ArrowUp)    my=-1;
    if (keys.KeyS||keys.ArrowDown)  my= 1;
    if (keys.KeyA||keys.ArrowLeft)  mx=-1;
    if (keys.KeyD||keys.ArrowRight) mx= 1;
    if (mx===0&&my===0) { mx=0;my=-1; }
    const dl=Math.sqrt(mx*mx+my*my)||1; dashDX=mx/dl; dashDY=my/dl;
    addParticles(player.x,player.y,'#88ffcc',10);
    questTick('dashes');
  }
  if (dashActive) {
    if (now-dashTimer<DASH_DUR) {
      player.x=Math.max(player.size,Math.min(W-player.size,player.x+dashDX*DASH_SPEED));
      player.y=Math.max(player.size,Math.min(H-player.size,player.y+dashDY*DASH_SPEED));
    } else { dashActive=false; }
  }

  // Player 1 movement (WASD)
  if (!dashActive) {
    const spd=player.speed;
    if (keys.KeyW && player.y>player.size)    player.y-=spd;
    if (keys.KeyS && player.y<H-player.size)  player.y+=spd;
    if (keys.KeyA && player.x>player.size)    player.x-=spd;
    if (keys.KeyD && player.x<W-player.size)  player.x+=spd;
  }


  // Hazards
  hazards.forEach((h,hi) => {
    if (h.type==='portal') h.angle=(h.angle||0)+0.04;
    const inH = distCC(player,h) < h.r+player.size;
    if (!inH) return;
    if (h.type==='swamp' && !dashActive) {
      player.x += (player.x < W/2 ? 0.5 : -0.5);
      player.y += (player.y < H/2 ? 0.5 : -0.5);
    }
    if (h.type==='spikes') { playerHit(player); return; }
    if (h.type==='portal') {
      const partner = hazards[h.partner];
      if (partner) { player.x=partner.x; player.y=partner.y; addParticles(player.x,player.y,'#cc88ff',12); }
    }
  });


  // Gem collection helper
  function collectGem(gobj, value, effect) {
    if (!gobj.active) return;
    const target = player;
    if (magnetUntil>now) {
      const dx=target.x-gobj.x,dy=target.y-gobj.y,d=Math.sqrt(dx*dx+dy*dy)||1;
      gobj.x+=dx/d*8; gobj.y+=dy/d*8;
    }
    const p1hit = distCC(player,gobj)<player.size+gobj.size;
    if (!p1hit) return;
    gobj.active=false; gobj.lastSpawn=now;
    addParticles(gobj.x,gobj.y,'#ffee88',10);
    if (effect) effect();
    if (value>0) {
      score+=value; scoreEl.textContent='Очки: '+score;
      questTick('gems'); questTick('score', 0);
      combo++; comboTimer=now;
      if (combo>=COMBO_BONUS) { comboEl.style.display='block'; comboEl.textContent=`🔥 КОМБО ×${Math.floor(combo/COMBO_BONUS)+1}`; questTick('combos'); }
    } else if (value<0) {
      score=Math.max(0,score+value); scoreEl.textContent='Очки: '+score;
      combo=0; comboEl.style.display='none';
    }
    if (value>0) {
      const spawnEvery = UPG.spawn.bought ? 5 : 3;
      if (score % spawnEvery===0) spawnMonster();
      if (score>=20&&!bossSpawned) { spawnBouncer(); bossSpawned=true; }
      if (score===30) spawnSniper();
      if (score===40) spawnInvisible();
      if (score===15) spawnSplitter();
      if (score%25===0&&score>0) spawnMiniBoss();
      if (currentMode==='normal'&&score>=WIN_SCORE) { winGame(); return; }
    }
  }

  // Main green gem
  if (gem.active) {
    if (magnetUntil>now) {
      const dx=player.x-gem.x,dy=player.y-gem.y,d=Math.sqrt(dx*dx+dy*dy)||1;
      gem.x+=dx/d*8; gem.y+=dy/d*8;
    }
    const p1h=distCC(player,gem)<player.size+gem.size;
    if (p1h) {
      gem.active=false;
      addParticles(gem.x,gem.y,'#00ff88',8);
      score++; scoreEl.textContent='Очки: '+score;
      questTick('gems'); questTick('score', 0);
      combo++; comboTimer=now;
      if (combo>=COMBO_BONUS) { comboEl.style.display='block'; comboEl.textContent=`🔥 КОМБО ×${Math.floor(combo/COMBO_BONUS)+1}`; }
      spawnGem();
      const spawnEvery=UPG.spawn.bought?5:3;
      if (score%spawnEvery===0) spawnMonster();
      if (score>=20&&!bossSpawned) { spawnBouncer(); bossSpawned=true; }
      if (score===30) spawnSniper();
      if (score===40) spawnInvisible();
      if (score===15) spawnSplitter();
      if (score%25===0&&score>0) spawnMiniBoss();
      if (currentMode==='normal'&&score>=WIN_SCORE) { winGame(); return; }
    }
  }

  collectGem(goldGem,3,null);
  collectGem(poisGem,-2,()=>{ statusEl.textContent='☠ Яд! -2 очка'; setTimeout(()=>statusEl.textContent='',2000); });
  collectGem(iceGem,1,()=>{ freezeUntil=now+4000; addParticles(player.x,player.y,'#aaddff',16); statusEl.textContent='❄ Заморозка!'; setTimeout(()=>statusEl.textContent='',4000); });
  collectGem(magnetGemObj,1,()=>{ magnetUntil=now+10000; statusEl.textContent='🧲 Магнит 10с'; setTimeout(()=>statusEl.textContent='',10000); });

  // Blue orb
  if (blueOrb.active && distCC(player,blueOrb)<player.size+blueOrb.size) {
    blueOrb.active=false; blueOrb.lastSpawnTime=now;
    if (monsters.length>0) {
      let idx=0,best=Infinity;
      monsters.forEach((m,i)=>{ const d=dist2(m,player); if(d<best){best=d;idx=i;} });
      addParticles(monsters[idx].x,monsters[idx].y,'#3399ff',12);
      if (monsters[idx].type==='splitter') {
        const m=monsters[idx];
        for(let k=0;k<2;k++) monsters.push({type:'chaser',x:m.x+(k?20:-20),y:m.y,size:14,speed:m.speed*1.2,vx:0,vy:0,color:'#ff44bb',hp:1});
      }
      monsters.splice(idx,1);
      questTick('orb_kills');
      monsterKillCoins++;
    }
  }

  // Projectiles
  for (let i=projectiles.length-1;i>=0;i--) {
    const p=projectiles[i];
    p.x+=p.vx; p.y+=p.vy; p.life--;
    if (p.life<=0||p.x<0||p.x>W||p.y<0||p.y>H) { projectiles.splice(i,1); continue; }
    if (distCC(p,player)<player.size+p.size) { projectiles.splice(i,1); playerHit(player); return; }
  }

  // Monsters AI
  const frozen = freezeUntil>now;
  resolveMonsterCollisions();
  for (let i=monsters.length-1;i>=0;i--) {
    const m=monsters[i];
    if (frozen && m.type!=='miniboss') { m.vx*=0.8; m.vy*=0.8; continue; }

    // Target: decoy > nearest player
    let tgt = player;
    if (decoy) tgt = decoy;

    if (m.type==='bouncer') {
      m.x+=m.vx; m.y+=m.vy;
      if (m.x-m.size<0||m.x+m.size>W) { m.vx=-m.vx; m.x=Math.max(m.size,Math.min(W-m.size,m.x)); }
      if (m.y-m.size<0||m.y+m.size>H) { m.vy=-m.vy; m.y=Math.max(m.size,Math.min(H-m.size,m.y)); }
    } else if (m.type==='sniper') {
      const dx=tgt.x-m.x, dy=tgt.y-m.y, d=Math.sqrt(dx*dx+dy*dy)||1;
      m.vx+=(dx/d*m.speed - m.vx)*0.05;
      m.vy+=(dy/d*m.speed - m.vy)*0.05;
      m.x+=m.vx; m.y+=m.vy;
      m.shootTimer=(m.shootTimer||0)+16;
      if (m.shootTimer>m.shootInterval) {
        m.shootTimer=0;
        const pdx=tgt.x-m.x,pdy=tgt.y-m.y,pd=Math.sqrt(pdx*pdx+pdy*pdy)||1;
        projectiles.push({x:m.x,y:m.y,vx:pdx/pd*5,vy:pdy/pd*5,size:7,life:120,color:'#ff6600'});
      }
    } else {
      const dx=tgt.x-m.x, dy=tgt.y-m.y, d=Math.sqrt(dx*dx+dy*dy)||1;
      m.vx+=(dx/d*m.speed - m.vx)*0.08;
      m.vy+=(dy/d*m.speed - m.vy)*0.08;
      m.x+=m.vx; m.y+=m.vy;
    }

    // Kill if off screen (non-arena)
    if (currentMode!=='arena') {
      if (m.x<-150||m.x>W+150||m.y<-150||m.y>H+150) { monsters.splice(i,1); continue; }
    }

    // Invisible: only visible near player
    if (m.type==='invisible') {
      const vis = distCC(m,player)<150;
      if (!vis) continue;
    }

    // Hit player 1
    const hitP1 = distCC(m,player)<m.size+player.drawSize && (player.alive!==false);
    if (hitP1) { m.hitMonster=m; player.hitMonster=m; playerHit(player); if(!gameRunning)return; }

  }

  // Particles
  for (let i=particles.length-1;i>=0;i--) {
    const p=particles[i];
    p.x+=p.vx; p.y+=p.vy; p.life-=0.035;
    if (p.life<=0) particles.splice(i,1);
  }

  // HUD cooldowns
  const now2=performance.now();
  const drem=DASH_CD-(now2-dashCooldown);
  dashCdEl.textContent = drem>0 ? `⚡ Рывок: ${(drem/1000).toFixed(1)}с` : '⚡ Рывок: готов';
  const trem=TP_CD-(now2-tpCooldown);
  tpCdEl.textContent   = trem>0 ? `🌀 Телепорт: ${(trem/1000).toFixed(1)}с` : '🌀 Телепорт: готов';
}

// ═══ DRAW ════════════════════════════════════════════════════════
function drawPlayer(p, color) {
  const ds = p===player ? player.drawSize : p.size;
  if (shieldActive && p===player) {
    ctx.shadowBlur=20; ctx.shadowColor='#00ccff';

}

    ctx.fillRect(0,0,W,H);

    grad.addColorStop(0,'rgba(26,60,26,1)');
    grad.addColorStop(0.7,'rgba(26,60,26,0.85)');
    grad.addColorStop(1,'rgba(3,10,3,0)');
    ctx.fillStyle=grad; ctx.beginPath(); ctx.arc(player.x,player.y,NIGHT_RADIUS,0,Math.PI*2); ctx.fill();
  } else {
    ctx.fillStyle='#1a3c1a'; ctx.fillRect(0,0,W,H);
    ctx.fillStyle='#0f2a0f';
    trees.forEach(t=>{ ctx.beginPath(); ctx.arc(t.x,t.y,t.r,0,Math.PI*2); ctx.fill(); });
  }

  // Arena boundary
  if (currentMode==='arena'&&arenaRadius>0) {
    ctx.strokeStyle='rgba(255,80,0,0.6)'; ctx.lineWidth=4;
    ctx.setLineDash([12,8]);
    ctx.beginPath(); ctx.arc(W/2,H/2,arenaRadius,0,Math.PI*2); ctx.stroke();
    ctx.setLineDash([]);
    ctx.save();
    ctx.beginPath(); ctx.rect(0,0,W,H);
    ctx.arc(W/2,H/2,arenaRadius,0,Math.PI*2,true);
    ctx.fillStyle='rgba(0,0,0,0.35)'; ctx.fill(); ctx.restore();
  }

  // Hazards
  hazards.forEach(h => {
    ctx.save();
    if (h.type==='swamp') {
      ctx.fillStyle='rgba(20,80,20,0.7)';
      ctx.beginPath(); ctx.ellipse(h.x,h.y,h.r*1.4,h.r*0.8,0,0,Math.PI*2); ctx.fill();
      ctx.fillStyle='rgba(0,200,50,0.3)';
      ctx.font='16px serif'; ctx.textAlign='center'; ctx.fillText('🌿',h.x,h.y+5);
    } else if (h.type==='spikes') {
      ctx.strokeStyle='#cc4400'; ctx.lineWidth=2;
      for (let k=0;k<8;k++) {
        const a=k/8*Math.PI*2;
        ctx.beginPath(); ctx.moveTo(h.x,h.y);
        ctx.lineTo(h.x+Math.cos(a)*h.r,h.y+Math.sin(a)*h.r); ctx.stroke();
      }
      ctx.fillStyle='#882200';
      ctx.beginPath(); ctx.arc(h.x,h.y,h.r*0.4,0,Math.PI*2); ctx.fill();
    } else if (h.type==='portal') {
      ctx.translate(h.x,h.y); ctx.rotate(h.angle||0);
      const pg=ctx.createRadialGradient(0,0,0,0,0,h.r);
      pg.addColorStop(0,'rgba(150,0,255,0.9)'); pg.addColorStop(1,'rgba(50,0,150,0)');

      ctx.strokeStyle='#cc44ff'; ctx.lineWidth=2;
      ctx.beginPath(); ctx.arc(0,0,h.r,0,Math.PI*2); ctx.stroke();

    ctx.lineTo(gem.x+gem.size*0.7, gem.y+gem.size*0.5);
    ctx.lineTo(gem.x-gem.size*0.7, gem.y+gem.size*0.5);
    ctx.closePath(); ctx.fill();
  }

    if (!g.active) return;
    ctx.fillStyle=color;
    ctx.beginPath();
    if (shape==='diamond') {
      ctx.moveTo(g.x,g.y-g.size); ctx.lineTo(g.x+g.size,g.y);
      ctx.lineTo(g.x,g.y+g.size); ctx.lineTo(g.x-g.size,g.y); ctx.closePath();
    } else {
      ctx.arc(g.x,g.y,g.size,0,Math.PI*2);
    }
    ctx.fill();
  }

  drawSpecGem(poisGem,'#aa00ff');
  drawSpecGem(iceGem,'#aaddff');

  monsters.forEach(m => {

      const vis = distCC(m,player)<150;
      if (!vis) return;
    }
    ctx.fillStyle = m.color;
    ctx.shadowBlur=8; ctx.shadowColor=m.color;
    ctx.beginPath(); ctx.arc(m.x,m.y,m.size,0,Math.PI*2); ctx.fill();
    ctx.shadowBlur=0;
    // Eyes
    ctx.fillStyle='white';
    ctx.beginPath(); ctx.arc(m.x-m.size*0.28,m.y-m.size*0.2,m.size*0.22,0,Math.PI*2); ctx.fill();
    ctx.beginPath(); ctx.arc(m.x+m.size*0.28,m.y-m.size*0.2,m.size*0.22,0,Math.PI*2); ctx.fill();

    ctx.beginPath(); ctx.arc(m.x-m.size*0.27,m.y-m.size*0.18,m.size*0.11,0,Math.PI*2); ctx.fill();
    ctx.beginPath(); ctx.arc(m.x+m.size*0.27,m.y-m.size*0.18,m.size*0.11,0,Math.PI*2); ctx.fill();
    // Miniboss HP bar
    if (m.type==='miniboss') {
      ctx.fillStyle='#333'; ctx.fillRect(m.x-m.size,m.y-m.size-10,m.size*2,6);
      ctx.fillStyle='#ff4444'; ctx.fillRect(m.x-m.size,m.y-m.size-10,m.size*2*(m.hp/3),6);
    }
  });

  projectiles.forEach(p => {
    ctx.fillStyle=p.color;
    ctx.shadowBlur=8; ctx.shadowColor=p.color;
    ctx.beginPath(); ctx.arc(p.x,p.y,p.size,0,Math.PI*2); ctx.fill();
    ctx.shadowBlur=0;
  });

  drawPlayer(player, player.color);

  // Night mask overlay
  if (isNight) {
    ctx.save();
    ctx.beginPath(); ctx.rect(0,0,W,H);
    ctx.arc(player.x,player.y,NIGHT_RADIUS,0,Math.PI*2,true);

  }

  // Particles
  particles.forEach(p => {
    ctx.globalAlpha=p.life;
    ctx.fillStyle=p.color;
    ctx.beginPath(); ctx.arc(p.x,p.y,4,0,Math.PI*2); ctx.fill();
  });
  ctx.globalAlpha=1;

  // Freeze overlay
  if (freezeUntil>performance.now()) {
    ctx.fillStyle='rgba(150,220,255,0.08)'; ctx.fillRect(0,0,W,H);
  }

  // Minimap
  drawMinimap();
}

// Seed: current date string → deterministic daily selection
function seededRand(seed, n) {
  let x = Math.sin(seed + n) * 43758.5453123;
  return x - Math.floor(x);
}


// Pick 3 quests for today deterministically
function getTodayQuests() {
  const seed = todaySeed();
  const pool = [...QUEST_POOL];
  const chosen = [];
  const used = new Set();
  for (let i = 0; chosen.length < 3 && i < 100; i++) {
    const idx = Math.floor(seededRand(seed, i) * pool.length);
    if (!used.has(idx)) { used.add(idx); chosen.push({...pool[idx]}); }
  }
  return chosen;
}

// Quest progress storage (daily persistence)

function loadQuestState() {
  try {
    const saved = safeSS.getItem(QUEST_KEY);
    if (saved) { questState = JSON.parse(saved); return; }
  } catch(e) {}
  questState = getTodayQuests().map(q => ({...q, progress: 0, claimed: false}));
  saveQuestState();
}
function saveQuestState() {
  try { safeSS.setItem(QUEST_KEY, JSON.stringify(questState)); } catch(e) {}
}

// questSessionStart declared near top

function questTick(type, amount=1) {
  if (!gameRunning || isPaused) return;
  if (type !== 'score') qTrack[type] = (qTrack[type] || 0) + amount;
  qTrack.score = score; // always sync
  // Update progress
  let updated = false;
  questState.forEach(q => {
    if (q.claimed) return;
    let prog = 0;
    if (q.type === 'time') {
      prog = Math.round((performance.now() - questSessionStart) / 1000);
    } else {
      prog = qTrack[q.type] || 0;
    }
    const newProg = Math.min(prog, q.target);
    if (newProg !== q.progress) { q.progress = newProg; updated = true; }
  });
  if (updated) { saveQuestState(); updateQuestHUD(); }
}

function questTimeTick() {
  if (!gameRunning || isPaused || !questSessionStart) return;
  questState.forEach(q => {
    if (q.claimed || q.type !== 'time') return;
    q.progress = Math.min(Math.round((performance.now() - questSessionStart) / 1000), q.target);
  });
  saveQuestState();
  updateQuestHUD();
}

function claimQuest(idx) {
  const q = questState[idx];
  if (!q || q.claimed || q.progress < q.target) return;
  q.claimed = true;
  score += q.reward;
  scoreEl.textContent = 'Очки: ' + score;
  earnCoins(Math.ceil(q.reward / 2), 'квест');
  saveQuestState();
  buildQuestHTML();
  updateQuestHUD();
  addParticles(player.x, player.y, '#ffdd44', 20);
}

function buildQuestHTML() {
  const list = document.getElementById('quest-list');
  list.innerHTML = '';
  questState.forEach((q, i) => {
    const pct = Math.min(100, Math.round(q.progress / q.target * 100));
    const done = q.progress >= q.target;
    const div = document.createElement('div');
    div.className = 'qi' + (done ? ' done' : '');
    div.innerHTML = `
      <div class="q-ic">${q.ic}</div>
      <div class="q-inf">
        <div class="q-nm">${q.nm}</div>
        <div class="q-bar-wrap"><div class="q-bar" style="width:${pct}%"></div></div>
        <div class="q-prog">${q.desc} — ${q.progress}/${q.target}</div>
      </div>
      <div class="q-rew">
        +${q.reward} очков<br>
        <button class="q-claim" onclick="claimQuest(${i})" ${(!done || q.claimed) ? 'disabled' : ''}>
          ${q.claimed ? '✓ Получено' : done ? 'Забрать' : '...'}
        </button>
      </div>`;
    list.appendChild(div);
  });
}

function updateQuestHUD() {
  const hud = document.getElementById('quest-hud');
  const active = questState.find(q => !q.claimed && q.progress < q.target);
  if (!active || !gameRunning) { hud.textContent = ''; return; }
  const pct = Math.round(active.progress / active.target * 100);
  hud.textContent = `📋 ${active.nm}: ${active.progress}/${active.target}`;
}
